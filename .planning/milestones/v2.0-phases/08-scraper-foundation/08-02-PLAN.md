This plan makes the endpoint return 200 with data from the pre-generated `scores-by-tohop.json` file, so the app stays functional with slightly stale data.

```typescript
if (err instanceof Error && err.message === 'DB_TIMEOUT') {
  const filePath = join(process.cwd(), 'public/data/scores-by-tohop.json');
  const raw = await readFile(filePath, 'utf-8');
  const allData: Record<string, CutoffDataRow[]> = JSON.parse(raw);
  const fallbackRows = allData[tohop] ?? [];
  const results = recommend(
    { tohop_code: tohop, total_score: totalScore },
    fallbackRows
  );
  const distinctYears = [...new Set(fallbackRows.map(r => r.year))].sort((a, b) => b - a);
  return Response.json({
    data: results,
    meta: { count: results.length, years_available: distinctYears, fallback: true },
  });
}
```

**Key differences from the normal path:**
- `meta.fallback: true` is added to the response so the frontend can surface a "using cached data" warning
- Uses `readFile` from `fs/promises` (async, not readFileSync) — aligns with FIX-08 pattern
- Falls back to empty array `[]` if the tohop code has no data in the static file
- Still calls `recommend()` on the fallback data — same algorithm, different data source

**Do NOT change:**
- The validation logic (lines 11-25) — params are validated before DB call, so 400s still work during fallback
- The `throw err` at line 85 — non-timeout errors should still bubble up
- The normal DB path (lines 28-78) — only the catch block changes

```typescript
// Mock withTimeout to always throw DB_TIMEOUT

      source_url: 'https://example.com',

  D01: [],

vi.mock('../../lib/db', () => ({ db: {} }));

import { GET } from '../../app/api/recommend/route';
import { NextRequest } from 'next/server';

describe('/api/recommend fallback', () => {
  it('returns 200 with fallback: true when DB times out', async () => {
    const req = new NextRequest('http://localhost/api/recommend?tohop=A00&score=25.0');
    const res = await GET(req);
    expect(res.status).toBe(200);
    const body = await res.json();
    expect(body.meta.fallback).toBe(true);
    expect(body.data).toBeDefined();
    expect(Array.isArray(body.data)).toBe(true);
  });

  it('returns empty results array for tohop with no fallback data', async () => {
    const req = new NextRequest('http://localhost/api/recommend?tohop=B01&score=25.0');
    const res = await GET(req);
    expect(res.status).toBe(200);
    const body = await res.json();
    expect(body.meta.fallback).toBe(true);
    expect(body.data).toEqual([]);
  });

  it('still returns 400 for invalid params even during DB outage', async () => {
    const req = new NextRequest('http://localhost/api/recommend?tohop=INVALID&score=25.0');
    const res = await GET(req);
    expect(res.status).toBe(400);
  });
});
```
