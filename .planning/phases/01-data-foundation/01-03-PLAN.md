    2. Create `lib/scraper/registry.ts`:
       ```typescript
       import { readFileSync } from 'fs';
       import { resolve } from 'path';
       import { ScraperAdapter } from './types';

       interface RegistryEntry {
         id: string;
         adapter: string;
         url: string;
         static_verified: boolean;
       }

       interface ResolvedEntry {
         id: string;
         adapter: ScraperAdapter;
         url: string;
       }

       export async function loadRegistry(): Promise<ResolvedEntry[]> {
         const configPath = resolve(process.cwd(), 'scrapers.json');
         const entries: RegistryEntry[] = JSON.parse(readFileSync(configPath, 'utf-8'));
         const resolved: ResolvedEntry[] = [];

         for (const entry of entries) {
           if (!entry.static_verified) {
             console.warn(`[registry] Skipping ${entry.id} — static_verified is false. Manually verify the page is static HTML before enabling.`);
             continue;
           }
           const mod = await import(`./adapters/${entry.adapter}`);
           // Each adapter module exports a default or named adapter object conforming to ScraperAdapter
           const adapter: ScraperAdapter = mod.default ?? mod[`${entry.adapter}Adapter`];
           resolved.push({ id: entry.id, adapter, url: entry.url });
         }

         return resolved;
       }
       ```

       **University adapters (bka.ts, kha.ts, nth.ts, gha.ts, dcn.ts):** Each follows this template:
       ```typescript
       import * as cheerio from 'cheerio';
       import { fetchHTML } from '../fetch';
       import { RawRow, ScraperAdapter } from '../types';

       export const [uni]Adapter: ScraperAdapter = {
         id: '[CODE]',
         async scrape(url: string): Promise<RawRow[]> {
           const html = await fetchHTML(url);
           const $ = cheerio.load(html);
           const rows: RawRow[] = [];

           // Semantic: find table with header containing cutoff score keywords
           $('table').each((_, table) => {
             const headers = $(table).find('th, thead td').map((_, el) => $(el).text().trim()).get();
             const scoreIdx = headers.findIndex(h =>
               h.includes('Diem chuan') || h.includes('Diem') || h.includes('diem'));
             const tohopIdx = headers.findIndex(h =>
               h.includes('To hop') || h.includes('Khoi') || h.includes('to hop'));
             const majorIdx = headers.findIndex(h =>
               h.includes('Ma nganh') || h.includes('Nganh') || h.includes('ma nganh'));

             if (scoreIdx === -1) return; // Not a cutoff table

             $(table).find('tbody tr').each((_, tr) => {
               const cells = $(tr).find('td').map((_, td) => $(td).text().trim()).get();
               if (cells.length === 0) return;
               rows.push({
                 university_id: '[CODE]',
                 major_raw: cells[majorIdx] ?? '',
                 tohop_raw: cells[tohopIdx] ?? '',
                 year: new Date().getFullYear() - 1, // Previous year's scores
                 score_raw: cells[scoreIdx] ?? '',
                 source_url: url,
               });
             });
           });

           // Minimum-rows assertion (Pitfall 4)
           if (rows.length === 0) {
             throw new Error(`[CODE] adapter returned 0 rows — possible JS rendering or layout change at ${url}`);
           }

           return rows;
         },
       };
       ```
       Replace `[CODE]` and `[uni]` for each university. Add university-specific URL path comments as TODOs (the exact cutoff page URL needs manual verification — the base URL from uni_list_examples.md is the homepage, not the cutoff page).
