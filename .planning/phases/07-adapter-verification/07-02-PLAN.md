```typescript
/**
 * DCN Adapter -- Truong Dai hoc Cong nghiep Ha Noi (HaUI)
 *
 * Rendering: JS-rendered (Playwright required)
 * URL: https://tuyensinh.haui.edu.vn/diem-chuan-trung-tuyen-dai-hoc
 * Pattern: Playwright reference adapter -- use as template for other JS-rendered pages
 *
 * University: Truong Dai hoc Cong nghiep Ha Noi
 * Ministry code: DCN
 * Homepage: https://www.haui.edu.vn/
 */

import { chromium } from 'playwright';
import * as cheerio from 'cheerio';
import { RawRow, ScraperAdapter } from '../types';

export const dcnAdapter: ScraperAdapter = {
  id: 'DCN',
  async scrape(url: string): Promise<RawRow[]> {
    const browser = await chromium.launch({ headless: true });
    let html: string;
    try {
      const page = await browser.newPage();
      await page.setExtraHTTPHeaders({
        'User-Agent': 'UniSelectBot/1.0 (educational; open source)',
      });
      await page.goto(url, { waitUntil: 'networkidle', timeout: 30000 });
      await page.waitForSelector('table', { timeout: 15000 });
      html = await page.content();
    } finally {
      await browser.close();
    }

    const $ = cheerio.load(html);
    const rows: RawRow[] = [];
    const year = new Date().getFullYear() - 1;

    $('table').each((_, table) => {
      const allRows = $(table).find('tr');
      if (allRows.length < 2) return;

      let headers: string[];
      const thHeaders = $(table).find('th, thead td').map((_, el) => $(el).text().trim().toLowerCase()).get();
      if (thHeaders.length > 0) {
        headers = thHeaders;
      } else {
        headers = $(allRows[0]).find('td').map((_, el) => $(el).text().trim().toLowerCase()).get();
      }

      const scoreIdx = headers.findIndex(
        (h) => h.includes('điểm chuẩn') || h.includes('diem chuan') || h.includes('điểm trúng tuyển')
      );
      const codeIdx = headers.findIndex(
        (h) => h.includes('mã ngành') || h.includes('ma nganh') || h.includes('ngành') || h.includes('nganh')
      );
      const tohopIdx = headers.findIndex(
        (h) => h.includes('tổ hợp') || h.includes('to hop') || h.includes('khối') || h.includes('khoi')
      );

      if (scoreIdx === -1) return;

      const dataRows = thHeaders.length > 0 ? $(table).find('tbody tr') : allRows.slice(1);
      dataRows.each((_, tr) => {
        const cells = $(tr).find('td').map((_, td) => $(td).text().trim()).get();
        if (cells.length < 3) return;

        const majorCode = codeIdx !== -1 ? cells[codeIdx] : '';
        const scoreRaw = cells[scoreIdx] ?? '';
        if (!majorCode || !scoreRaw) return;
        if (!/^\d/.test(majorCode)) return;

        rows.push({
          university_id: 'DCN',
          major_raw: majorCode,
          tohop_raw: tohopIdx !== -1 ? cells[tohopIdx] : '',
          year,
          score_raw: scoreRaw,
          source_url: url,
        });
      });
    });

    if (rows.length === 0) {
      throw new Error(`DCN adapter returned 0 rows -- possible rendering failure at ${url}`);
    }
    return rows;
  },
};
```

```typescript
/**
 * GHA Adapter -- Truong Dai hoc Giao thong van tai (UTC)
 *
 * Rendering: JPEG images (PaddleOCR required)
 * URL: https://tuyensinh.utc.edu.vn/?q=tin-tuyen-sinh/diem-trung-tuyen-dai-hoc-he-chinh-quy-nam-2025
 * Pattern: PaddleOCR reference adapter -- use as template for other image-based pages
 *
 * University: Truong Dai hoc Giao thong van tai
 * Ministry code: GHA
 * Homepage: https://utc.edu.vn/
 */

import { execSync } from 'child_process';
import { writeFileSync, readFileSync, unlinkSync } from 'fs';
import { tmpdir } from 'os';
import { join } from 'path';
import { fetchHTML } from '../fetch';
import * as cheerio from 'cheerio';
import { RawRow, ScraperAdapter } from '../types';

export const ghaAdapter: ScraperAdapter = {
  id: 'GHA',
  async scrape(url: string): Promise<RawRow[]> {
    // Step 1: Fetch announcement page to find image URLs
    const html = await fetchHTML(url);
    const $ = cheerio.load(html);
    const imageUrls: string[] = [];
    $('img').each((_, el) => {
      const src = $(el).attr('src') ?? '';
      if (src.match(/diem.*\.(jpg|jpeg|png)/i) || src.match(/page-\d+\.(jpg|jpeg|png)/i)) {
        // Resolve relative URLs
        const absolute = src.startsWith('http') ? src : new URL(src, url).href;
        imageUrls.push(absolute);
      }
    });

    if (imageUrls.length === 0) {
      throw new Error(`GHA adapter found 0 score images at ${url}`);
    }

    const rows: RawRow[] = [];
    const year = new Date().getFullYear() - 1;

    for (const imgUrl of imageUrls) {
      // Step 2: Download image to temp file
      const res = await fetch(imgUrl);
      if (!res.ok) continue;
      const buffer = Buffer.from(await res.arrayBuffer());
      const tempImg = join(tmpdir(), `gha_score_${Date.now()}.jpg`);
      const tempOut = join(tmpdir(), `gha_ocr_${Date.now()}.json`);

      try {
        writeFileSync(tempImg, buffer);

        // Step 3: Shell out to PaddleOCR
        execSync(`python3 scripts/ocr_table.py "${tempImg}" "${tempOut}"`, {
          timeout: 120000,
          cwd: process.cwd(),
        });

        // Step 4: Parse OCR output
        const ocrResult = JSON.parse(readFileSync(tempOut, 'utf-8'));
        const lines: string[] = ocrResult.lines ?? [];

        // Parse lines into rows: look for patterns of [number, code, name, tohop, score]
        // OCR outputs text top-to-bottom, left-to-right within each line
        for (let i = 0; i < lines.length; i++) {
          const line = lines[i];
          // Look for major codes (7-digit numbers starting with 7)
          const majorMatch = line.match(/^(7\d{6})/);
          if (majorMatch) {
            // Next lines should be name, tohop, score
            const majorCode = majorMatch[1];
            const tohop = lines[i + 1] ?? '';
            const score = lines[i + 2] ?? '';
            if (/\d/.test(score)) {
              rows.push({
                university_id: 'GHA',
                major_raw: majorCode,
                tohop_raw: tohop,
                year,
                score_raw: score.replace(',', '.'),
                source_url: url,
              });
            }
          }
        }
      } finally {
        try { unlinkSync(tempImg); } catch { /* ignore */ }
        try { unlinkSync(tempOut); } catch { /* ignore */ }
      }
    }

    if (rows.length === 0) {
      throw new Error(`GHA OCR adapter returned 0 rows from ${imageUrls.length} images at ${url}`);
    }
    return rows;
  },
};
```

**Create `scripts/ocr_table.py`:**
```python
#!/usr/bin/env python3
"""
PaddleOCR table text extraction helper.

Usage: python3 scripts/ocr_table.py <image_path> <output_json_path>

Reads an image file, runs PaddleOCR with Vietnamese language support,
and writes extracted text lines to a JSON file.

Output format: {"lines": ["line1", "line2", ...]}
"""

import sys
import json

if len(sys.argv) != 3:
    print("Usage: python3 ocr_table.py <image_path> <output_json_path>", file=sys.stderr)
    sys.exit(1)

image_path = sys.argv[1]
output_path = sys.argv[2]

from paddleocr import PaddleOCR

ocr = PaddleOCR(use_angle_cls=True, lang='vi', use_gpu=False, show_log=False)
result = ocr.ocr(image_path, cls=True)

lines = []
if result:
    for page in result:
        if page:
            for line in page:
                text = line[1][0]  # (bounding_box, (text, confidence))
                lines.append(text)

with open(output_path, 'w', encoding='utf-8') as f:
    json.dump({'lines': lines}, f, ensure_ascii=False)

print(f"OCR extracted {len(lines)} lines from {image_path}", file=sys.stderr)
```

**Create `docs/adapter-strategies.md`:**

```markdown
# Adapter Strategies

UniSelect scrapers handle three types of university cutoff score pages.
Each type has a reference adapter that serves as a template for new adapters.

## Strategy 1: Static HTML (Cheerio)

**When to use:** Page source contains `<table>` with score data visible in View Source (Ctrl+U).

**Reference adapter:** `lib/scraper/adapters/htc.ts` (HTC -- Hoc vien Tai chinh)

**How it works:**
1. `fetchHTML(url)` fetches raw HTML (handles encoding via chardet)
2. `cheerio.load(html)` parses the DOM
3. Find score table by semantic header text (e.g. "diem chuan", "THPT")
4. Extract rows into `RawRow[]`

**Dependencies:** cheerio (already installed)
**CI overhead:** None
**Verified adapters:** HTC, BVH (PTIT)

---

## Strategy 2: Playwright (JS-Rendered Pages)

**When to use:** Page shows score table in browser but View Source has no `<table>` or data.
Common with React/Vue SPAs, AJAX-loaded tables, or content behind JavaScript hydration.

**Reference adapter:** `lib/scraper/adapters/dcn.ts` (DCN -- Dai hoc Cong nghiep Ha Noi / HaUI)

**How it works:**
1. `chromium.launch({ headless: true })` starts headless browser
2. `page.goto(url, { waitUntil: 'networkidle' })` loads page with full JS execution
3. `page.waitForSelector('table')` waits for score table to appear in DOM
4. `page.content()` gets fully-rendered HTML
5. `cheerio.load(html)` and same table parsing as static adapter
6. `browser.close()` in `finally` block (CRITICAL -- leaked browsers kill CI)

**Dependencies:** playwright (devDependency), chromium browser binary
**CI overhead:** ~300MB download, ~30s per adapter execution
**CI setup:** `npx playwright install chromium --with-deps`

**Important rules:**
- Always use `try/finally` with `browser.close()`
- `waitUntil: 'networkidle'` is more reliable than `domcontentloaded` for SPA tables
- After `page.content()`, reuse cheerio parsing -- keeps logic uniform with static adapters
- The adapter still exports `ScraperAdapter` interface -- registry.ts needs no changes

---

## Strategy 3: PaddleOCR (Image-Based Pages)

**When to use:** Score data is published as JPEG/PNG images (scanned announcements).
Common with official "thong bao" (notification) documents.

**Reference adapter:** `lib/scraper/adapters/gha.ts` (GHA -- Dai hoc Giao thong van tai / UTC)

**How it works:**
1. `fetchHTML(url)` fetches announcement page
2. Find `<img>` tags matching score image patterns (e.g. `diem.*page-*.jpg`)
3. Download each image to temp file
4. `execSync('python3 scripts/ocr_table.py <img> <out>')` runs PaddleOCR
5. Parse JSON output (`{"lines": [...]}`) into `RawRow[]`
6. Clean up temp files in `finally` block

**Dependencies:** Python 3.11+, paddleocr pip package
**CI overhead:** ~200MB model download on first run, ~10-30s per image
**CI setup:**
```yaml
- uses: actions/setup-python@v5
  with:
    python-version: '3.11'
- run: pip install paddleocr
- run: python3 -c "from paddleocr import PaddleOCR; PaddleOCR(lang='vi', use_gpu=False)"
```

**Important rules:**
- PaddleOCR lazy-downloads models (~50-200MB) on first use -- warm up in CI
- Vietnamese diacritics on numbers/codes have HIGH accuracy; major names may have errors
- Always clean up temp files in `finally` block
- Validate OCR output manually before trusting row counts

---

## Adding a New Adapter
