**Update .github/workflows/scrape-low.yml:**

Replace the single-job scrape workflow with a matrix-sharded version. The updated workflow must:

1. Keep the same trigger (daily cron `0 2 * * *` + workflow_dispatch)
2. Define a matrix with `shard: [0, 1, 2, 3, 4, 5]` — 6 shards
3. Set `fail-fast: false` so one shard failure does not cancel the others
4. Pass `SHARD_INDEX: ${{ matrix.shard }}` and `SHARD_TOTAL: 6` as env vars to the scraper step
5. Keep the same Node.js 20 + npm ci + `npx tsx lib/scraper/run.ts` run command
6. Pass the same secrets (DATABASE_URL, SUPABASE_SERVICE_ROLE_KEY, GITHUB_RUN_ID)

```yaml
name: Scrape cutoff scores (low frequency)

on:
  schedule:
    - cron: '0 2 * * *'
  workflow_dispatch:

jobs:
  scrape:
    runs-on: ubuntu-latest
    strategy:
      matrix:
        shard: [0, 1, 2, 3, 4, 5]
      fail-fast: false
    steps:
      - uses: actions/checkout@v4
      - uses: actions/setup-node@v4
        with:
          node-version: '20'
          cache: 'npm'
      - run: npm ci
      - name: Run scraper shard ${{ matrix.shard }}/6
        run: npx tsx lib/scraper/run.ts
        env:
          DATABASE_URL: ${{ secrets.DATABASE_URL }}
          SUPABASE_SERVICE_ROLE_KEY: ${{ secrets.SUPABASE_SERVICE_ROLE_KEY }}
          GITHUB_RUN_ID: ${{ github.run_id }}
          SHARD_INDEX: ${{ matrix.shard }}
          SHARD_TOTAL: 6
```

**Create .github/workflows/scrape-peak.yml:**

Peak workflow runs 4× per day during July. Key design decisions:
- Cron: `0 1,7,13,19 * * *` — 01:00, 07:00, 13:00, 19:00 UTC (spread across day)
- Month constraint: `7` — July only (GitHub cron month field)
  - Actually GitHub Actions cron does NOT support month filtering natively. Use the `if: github.event.schedule == '...' ` pattern OR use a repository variable toggle.
  - Use repository variable approach: job condition `if: vars.PEAK_SCHEDULE_ENABLED == 'true'`
  - This lets the user toggle the peak schedule on/off from GitHub Settings > Variables without touching code
- Same matrix structure as scrape-low.yml (6 shards, fail-fast: false)
- workflow_dispatch always available for manual trigger (no condition on manual trigger)

```yaml
name: Scrape cutoff scores (peak frequency)

on:
  schedule:
    - cron: '0 1,7,13,19 * * *'   # 4x daily — enable in July via PEAK_SCHEDULE_ENABLED
  workflow_dispatch:                 # Manual trigger always works

jobs:
  scrape:
    # For scheduled runs: only execute when PEAK_SCHEDULE_ENABLED='true'
    # For manual workflow_dispatch: always execute
    if: github.event_name == 'workflow_dispatch' || vars.PEAK_SCHEDULE_ENABLED == 'true'
    runs-on: ubuntu-latest
    strategy:
      matrix:
        shard: [0, 1, 2, 3, 4, 5]
      fail-fast: false
    steps:
      - uses: actions/checkout@v4
      - uses: actions/setup-node@v4
        with:
          node-version: '20'
          cache: 'npm'
      - run: npm ci
      - name: Run scraper shard ${{ matrix.shard }}/6 (peak)
        run: npx tsx lib/scraper/run.ts
        env:
          DATABASE_URL: ${{ secrets.DATABASE_URL }}
          SUPABASE_SERVICE_ROLE_KEY: ${{ secrets.SUPABASE_SERVICE_ROLE_KEY }}
          GITHUB_RUN_ID: ${{ github.run_id }}
          SHARD_INDEX: ${{ matrix.shard }}
          SHARD_TOTAL: 6
```

Add a comment block at the top of the file:

**Expand adapter-contract.test.ts:**

The current file manually imports 6 adapters. With 78 adapters, manual imports are unmaintainable. Replace with a dynamic import approach that reads scrapers.json and imports each adapter module at test time.

New structure:

```typescript
import { describe, it, expect } from 'vitest';
import { readFileSync } from 'fs';
import { resolve } from 'path';

interface RegistryEntry {
  id: string;
  adapter: string;
  static_verified: boolean;
  note?: string;
}

const registry: RegistryEntry[] = JSON.parse(
  readFileSync(resolve(process.cwd(), 'scrapers.json'), 'utf-8')
);

describe('Adapter contract: all adapters conform to ScraperAdapter interface', () => {
  for (const entry of registry) {
    const adapterFile = `../../../lib/scraper/adapters/${entry.adapter}`;
    const exportName = `${entry.adapter}Adapter`;

    describe(`${exportName} (${entry.id})`, () => {
      let adapter: { id: string; scrape: unknown };

      beforeAll(async () => {
        const mod = await import(adapterFile);
        adapter = mod[exportName];
      });

      it('exports an object (not null/undefined)', () => {
        expect(adapter).toBeDefined();
        expect(adapter).not.toBeNull();
        expect(typeof adapter).toBe('object');
      });

      it('has an id property of type string', () => {
        expect(typeof adapter.id).toBe('string');
        expect(adapter.id.length).toBeGreaterThan(0);
      });

      it(`has id equal to "${entry.id}"`, () => {
        expect(adapter.id).toBe(entry.id);
      });

      it('has a scrape property of type function', () => {
        expect(typeof adapter.scrape).toBe('function');
      });
    });
  }
});
```

Note: `beforeAll` requires importing from vitest — add it to the import line.

The dynamic approach means: when a new adapter is added to scrapers.json, the contract test automatically covers it with zero test file changes.
