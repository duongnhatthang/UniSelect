```typescript
export const htcAdapter: ScraperAdapter = {
  id: 'HTC',
  async scrape(url: string): Promise<RawRow[]> {
    const html = await fetchHTML(url);
    const $ = cheerio.load(html);
    const rows: RawRow[] = [];
    const year = new Date().getFullYear() - 1;

    $('table').each((_, table) => {
      const allRows = $(table).find('tr');
      if (allRows.length < 2) return;

      // Headers: prefer <th>/<thead td>, fall back to first <tr><td>
      let headers: string[];
      const thHeaders = $(table).find('th, thead td').map((_, el) => $(el).text().trim().toLowerCase()).get();
      if (thHeaders.length > 0) {
        headers = thHeaders;
      } else {
        headers = $(allRows[0]).find('td').map((_, el) => $(el).text().trim().toLowerCase()).get();
      }

      const scoreIdx = headers.findIndex(h =>
        h.includes('điểm trúng tuyển') || h.includes('diem trung tuyen') ||
        h.includes('điểm chuẩn') || h.includes('diem chuan')
      );
      const codeIdx = headers.findIndex(h =>
        h.includes('mã ngành') || h.includes('ma nganh') ||
        h.includes('mã xét tuyển') || h.includes('ma xet tuyen')
      );

      if (scoreIdx === -1) return;

      allRows.slice(1).each((_, tr) => {
        const cells = $(tr).find('td').map((_, td) => $(td).text().trim()).get();
        if (cells.length < 3) return;
        const majorCode = codeIdx !== -1 ? cells[codeIdx] : '';
        const scoreRaw = cells[scoreIdx] ?? '';
        if (!majorCode || !scoreRaw) return;
        if (!/^\d/.test(majorCode)) return;

        rows.push({
          university_id: 'HTC',
          major_raw: majorCode,
          tohop_raw: 'A00',  // defaultTohop
          year,
          score_raw: scoreRaw,
          source_url: url,
        });
      });
    });

    if (rows.length === 0) {
      throw new Error(`HTC adapter returned 0 rows — possible JS rendering or layout change at ${url}`);
    }
    return rows;
  },
};
```

**1. Create `lib/scraper/factory.ts`:**

```typescript
import * as cheerio from 'cheerio';
import { fetchHTML } from './fetch';
import type { RawRow, ScraperAdapter } from './types';

export interface CheerioAdapterConfig {
  id: string;
  scoreKeywords: string[];    // e.g. ['điểm chuẩn', 'diem chuan', 'điểm trúng tuyển']
  majorKeywords: string[];    // e.g. ['mã ngành', 'ma nganh', 'mã xét tuyển']
  tohopKeywords?: string[];   // e.g. ['tổ hợp', 'to hop'] — omit for single-tohop
  defaultTohop?: string;      // e.g. 'A00' — used when tohopKeywords is omitted or tohop column not found
}
```

The `createCheerioAdapter(config: CheerioAdapterConfig): ScraperAdapter` function must:

a. Return `{ id: config.id, scrape: async (url) => { ... } }`

b. Inside `scrape`:
   - Call `fetchHTML(url)` and `cheerio.load(html)`
   - Set `const year = new Date().getFullYear() - 1;`
   - Iterate `$('table').each(...)`:
     - Find all `<tr>` rows; skip tables with < 2 rows
     - **Header detection (matching HTC pattern exactly):**
       - Try `$(table).find('th, thead td')` first
       - If empty, use first `<tr> td` cells as headers
       - Lowercase + trim all header strings
     - **Column index finding:**
       - `scoreIdx`: first header where ANY `config.scoreKeywords` entry is found via `.includes()`
       - `codeIdx`: first header where ANY `config.majorKeywords` entry is found via `.includes()`
       - `tohopIdx`: if `config.tohopKeywords` provided, first header where ANY keyword found; else `-1`
     - If `scoreIdx === -1`, skip this table (`return`)
     - **Row extraction (skip header row, process remaining):**
       - Get cells from each `<tr> td`, trim text
       - Skip rows with < 3 cells
       - `majorCode = codeIdx !== -1 ? cells[codeIdx] : ''`
       - `scoreRaw = cells[scoreIdx] ?? ''`
       - `tohopRaw = tohopIdx !== -1 ? cells[tohopIdx] : (config.defaultTohop ?? '')`
       - Skip if `!majorCode || !scoreRaw`
       - Skip if `!/^\d/.test(majorCode)` (section headers like "I", "II")
       - Push `RawRow` with `university_id: config.id`
   - After all tables: if `rows.length === 0`, throw `Error(\`${config.id} adapter returned 0 rows — possible JS rendering or layout change at ${url}\`)`
   - Return rows

**2. Update `lib/scraper/registry.ts`:**

Add `factory_config` support:

```typescript
import { readFileSync } from 'fs';
import { resolve } from 'path';
import type { ScraperAdapter } from './types';
import { createCheerioAdapter } from './factory';
import type { CheerioAdapterConfig } from './factory';

interface RegistryEntry {
  id: string;
  adapter: string;
  url: string;
  static_verified: boolean;
  note?: string;
  factory_config?: Omit<CheerioAdapterConfig, 'id'>;  // id comes from entry.id
}

interface ResolvedEntry {
  id: string;
  adapter: ScraperAdapter;
  url: string;
}

export async function loadRegistry(): Promise<ResolvedEntry[]> {
  const configPath = resolve(process.cwd(), 'scrapers.json');
  const entries: RegistryEntry[] = JSON.parse(readFileSync(configPath, 'utf-8'));
  const resolved: ResolvedEntry[] = [];

  for (const entry of entries) {
    if (!entry.static_verified) {
      console.warn(
        `[registry] Skipping ${entry.id} — static_verified is false.`
      );
      continue;
    }

    let adapter: ScraperAdapter;

    if (entry.factory_config) {
      // Use factory for cheerio-based adapters
      adapter = createCheerioAdapter({ id: entry.id, ...entry.factory_config });
    } else {
      // Dynamic import for non-cheerio adapters (GHA/PaddleOCR, Playwright)
      const mod = await import(`./adapters/${entry.adapter}`);
      adapter = mod.default ?? mod[`${entry.adapter}Adapter`];
    }

    resolved.push({ id: entry.id, adapter, url: entry.url });
  }

  return resolved;
}
```
